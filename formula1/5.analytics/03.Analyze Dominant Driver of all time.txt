with driver_matrics as (
SELECT  driver_name,
        sum(race_starts) as race_starts,
        sum(number_of_wins) as total_wins,
        sum(number_of_podiums) as total_podiums,
        sum(case when standing_position = 1 then 1 else 0 end) as total_championships
FROM formula1.gold.view_driver_standing
group by driver_name
having total_championships >= 1
)
select driver_name,
       race_starts,
       total_wins,
       total_podiums, 
       total_championships,
       (total_championships * 100) + (total_wins * 10) + (total_podiums * 3) as greatness_score
from driver_matrics
order by greatness_score desc
